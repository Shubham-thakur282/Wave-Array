class Solution{
    public:
    // arr: input array
    // n: size of array
    //Function to sort the array into a wave-like array.
    void convertToWave(int n, vector<int>a){
        
        // Your code here
        if(n%2==0)
        {    
            for(int i=0;i<n;i+=2)
            {
                int temp=a[i];
                a[i]=a[i+1];
                a[i+1]=temp;
            }
            for(int i=0;i<n;i++)
            cout<<a[i]<<" ";
        }
        else
        {
            for(int i=0;i<n-1;i+=2)
            {
                int temp=a[i];
                a[i]=a[i+1];
                a[i+1]=temp; 
            }
            for(int i=0;i<n;i++)
            cout<<a[i]<<" ";
        }
    }
};
